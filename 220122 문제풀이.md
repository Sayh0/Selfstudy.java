# 짝수의 합 구하기
```
import java.util.*;
public class Main {
	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		int inputNum = sc.nextInt();
		int sum = 0;
		for(int i = 0; i <= inputNum; i++) {
			if(i % 2 == 0 ) {
				sum = sum + i;
			}
			
		}
		System.out.println(sum);
	}
}
```

# N to M 2
```java
package sec01.exam01;
import java.util.*;
public class Main {
	public static void main(String[] args) {
	
		Scanner sc = new Scanner(System.in);
		int n = sc.nextInt();
		int m = sc.nextInt();
		int unit = 8;
		int cnt = 0;
		for (int i = n; i <= m; i++) {
			System.out.print(i + " ");
			cnt++;
			if( cnt % unit == 0) {
				System.out.printf("\n");
			}
		}
			
	}
}
```
