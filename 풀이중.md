import java.util.Scanner;
public class Main {
	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		int i = sc.nextInt();
		
		int standard=i;
		int j = 0; //차감용 변수
		for (j = 0; j < standard; j++) {
			System.out.println(i);
			i = i - j;
			System.out.println(j);
		}
	
	}
}
